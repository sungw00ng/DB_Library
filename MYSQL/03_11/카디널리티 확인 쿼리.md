```sql
SELECT 
    kcu.TABLE_NAME AS '자식테이블',
    kcu.COLUMN_NAME AS 'FK컬럼',
    kcu.REFERENCED_TABLE_NAME AS '부모테이블',
    kcu.REFERENCED_COLUMN_NAME AS '참조컬럼',
    IF(col.IS_NULLABLE = 'YES', '0..N (선택)', '1..N (필수)') AS '관계타입'
FROM INFORMATION_SCHEMA.KEY_COLUMN_USAGE kcu
JOIN INFORMATION_SCHEMA.COLUMNS col
    ON kcu.TABLE_SCHEMA = col.TABLE_SCHEMA
    AND kcu.TABLE_NAME = col.TABLE_NAME
    AND kcu.COLUMN_NAME = col.COLUMN_NAME
WHERE kcu.TABLE_SCHEMA = 'bookstore'
AND kcu.REFERENCED_TABLE_NAME IS NOT NULL;
```
