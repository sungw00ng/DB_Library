- member 슈퍼타입/서브타입 설계
```
member (공통 속성)
├── regular_member (일반회원 전용 속성) 8만
├── admin (관리자 전용 속성) 1만
└── seller (판매자 전용 속성) 1만

ALTER TABLE member ADD COLUMN member_type ENUM('REGULAR', 'ADMIN', 'SELLER');
ALTER TABLE member ADD COLUMN store_name VARCHAR(100); -- 판매자
ALTER TABLE member ADD COLUMN level INT; -- 관리자
ALTER TABLE member ADD COLUMN point INT DEFAULT 0; --일반회원

-- 무작위로 분배
UPDATE member
SET member_type = CASE
    WHEN member_id % 10 = 0 THEN 'ADMIN'
    WHEN member_id % 5 = 0 THEN 'SELLER'
    ELSE 'REGULAR'
END;
```
